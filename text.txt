name: Deploy to Raspberry Pi

on:
  pull_request:
    branches: [main, master]
  push:
    branches: [main, master]
  workflow_dispatch:

concurrency:
  group: frontend-deploy-${{ github.ref }}
  cancel-in-progress: true

env:
  REGISTRY: ghcr.io
  IMAGE_BASE: ${{ github.repository }}

jobs:
  build-frontend:
    runs-on: [self-hosted, linux, arm64, RPI5]
    permissions:
      contents: read
      packages: write
    environment: production

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3
      
      - name: Log in to GitHub Container Registry
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract Metadata
        id: meta
        uses: docker/metadata-action@v5
        with:
            images: ${{ env.REGISTRY }}/${{ env.IMAGE_BASE }}-frontend
            tags: |
                type=raw,value=${{ github.sha }}
                type=raw,value=latest

      - name: Build and push (Frontend)
        uses: docker/build-push-action@v5
        with:
          context: .
          platforms: linux/arm64
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=local,src=/home/ethan/.buildx-cache/finance
          cache-to: type=local,dest=/home/ethan/.buildx-cache/finance-new,mode=max
          provenance: false
          sbom: false

      - name: Rotate frontend local cache
        run: |
          rm -rf /home/ethan/.buildx-cache/frontend
          mv /home/ethan/.buildx-cache/frontend-new /home/ethan/.buildx-cache/frontend || true

  build-backend:
    runs-on: [self-hosted, Linux, ARM64, pi-builder, raspberrypi]
    permissions:
      contents: read
      packages: write
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Log in to GitHub Container Registry
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata (Backend)
        id: meta-backend
        uses: docker/metadata-action@v5
        with:
          images: ${{ env.REGISTRY }}/${{ env.IMAGE_BASE }}-backend
          tags: |
            type=raw,value=${{ github.sha }}
            type=raw,value=latest

      - name: Build and push (Backend)
        uses: docker/build-push-action@v5
        with:
          context: ./backend
          platforms: linux/arm64
          push: true
          tags: ${{ steps.meta-backend.outputs.tags }}
          labels: ${{ steps.meta-backend.outputs.labels }}
          cache-from: type=local,src=/home/ethan/.buildx-cache/backend
          cache-to: type=local,dest=/home/ethan/.buildx-cache/backend-new,mode=max
          provenance: false
          sbom: false

      - name: Rotate backend local cache
        run: |
          rm -rf /home/ethan/.buildx-cache/backend
          mv /home/ethan/.buildx-cache/backend-new /home/ethan/.buildx-cache/backend || true

  deploy:
    runs-on: ubuntu-latest
    needs: [build-frontend]
    permissions:
      contents: read
      packages: read
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Checkout Helm Charts
        uses: actions/checkout@v4
        with:
          repository: ethanlchristensen/bruh.kubeinfra
          token: ${{ secrets.GH_PAT }}

      - name: Tailscale
        uses: tailscale/github-action@v4
        with:
          authkey: ${{ secrets.TS_AUTHKEY }}

      - name: Set up kubectl
        uses: azure/setup-kubectl@v4

      - name: Set up Helm
        uses: azure/setup-helm@v4

      - name: Configure kubeconfig
        run: |
          mkdir -p $HOME/.kube
          echo "${{ secrets.KUBECONFIG }}" | base64 -d > $HOME/.kube/config
          chmod 600 $HOME/.kube/config
      - name: Deploy Backend
        run: |
          helm upgrade --install bruh-finance-backend app-chart \
          -f app-values/values-backend.yaml
          --set image=${{ env.REGISTRY }}/${{ env.IMAGE_BASE }}-frontend:${{ github.sha }} \
          -n my-apps

      - name: Deploy frontend
        run: |
           helm upgrade --install bruh-finance-frontend app-chart \
            -f app-values/values-frontend.yaml \
            --set image=${{ env.REGISTRY }}/${{ env.IMAGE_BASE }}-frontend:${{ github.sha }} \
            -n my-apps
